\begin{homeworkProblem}
\textit{(48 pts.)} Represent the following sentences in first-order logic, using a consistent vocabulary (which you must define):

\underline{\textbf{Definitions}}: $\textbf{Takes(x,y,z)}$: student x takes course y in semester z; $\textbf{Passes(x,y,z)}$: student x passes course y in semester z; $\textbf{Score(x,y)}$: The score of student x in course y; $\textbf{Buys(x,y,z)}$: person x buys y from z; $\textbf{Sells(x,y,z)}$: person x sells y to z; $\textbf{Shaves(x,y)}$: x shaves y;$\textbf{Citizen(x,y,z)}$: person x is a citizen of country y for reason z; $\textbf{Resident(x,y)}$: person x is a resident of country y; $\textbf{Fools(x,y,z)}$: x fools y for time z.

\begin{enumerate}[label=\alph*.]
    \item Some students took French in spring 2001. 
    
    \textbf{Answer:} $\exists x$  $Student(x)\land Takes(x, French, Spring2011) $
    
    \item Every student who takes French passes it.
    
    \textbf{Answer:} $\forall x$  $Student(x)\land Takes(x, French, Spring2011) \Rightarrow Passes(x, French, Spring2011) $
    \item Only one student took Greek in spring 2001.
    
    \textbf{Answer:} 
    
    $\exists x$, $Student(x)\land Takes(x, Greek, Spring2001)\land [\forall y$  $y\neq x$, $Student(y)]\Rightarrow \neg Takes(y, Greek, Spring2001)$
    \item The best score in Greek is always higher than the best score in French.
    
    \textbf{Answer:} $\exists x$  $\forall y$, $Score(x,Greek) > Score(y, French)$
    \item Every person who buys a policy is smart.
    
    \textbf{Answer:} $\forall x$  $Person(x) \land [\exists y,z$ $Policy(y)\land Buys(x,y,z)]\Rightarrow Smart(x)$
    \item No person buys an expensive policy.
    
    \textbf{Answer:} $\forall x, y,z$  $Person(x)\land Policy(y)\land Expensive(y)\Rightarrow \neg Buys(x,y,z)$
    \item There is an agent who sells policies only to people who are not insured.
    
    \textbf{Answer:} $\exists x$ $Agent(x)\land \{\forall y$  $Policy(y)\land [\forall z$ $Sells(x, y, z)]\}\Rightarrow Person(z)\land \neg Insured(z)$
    \item There is a barber who shaves all men in town who do not shave themselves.
    
    \textbf{Answer:} $\exists b$ $\forall m$ $Barber(x)\land Man(m)\land \neg Shaves(b, b)\Rightarrow Shaves(b, m)$
    \item A person born in the UK, each of whose parents is a UK citizen or a UK resident, is a UK citizen by birth.
    
    \textbf{Answer:} 
    
    $\forall x$ $Person(x)\land Born(x, UK)\land [\forall y$ $Parent(y, x)\Rightarrow Citizen(y, UK, r)\lor Resident(y, UK)]\Rightarrow Citizen(x, UK, birth)$
    \item A person born outside the UK, one of whose parents is a UK citizen by birth, is a UK citizen by descent.
    
    \textbf{Answer:} $\forall x$ $Person(x)\land \neg Born(x,UK)\land [\exists y$ $Parent(y,x)\land Citizen(y, UK, birth)] \Rightarrow Citizen(x,UK,descent)$
    \item Politicians can fool some of the people all of the time, and they can fool all of the people some of the time, but they canâ€™t fool all of the people all of the time.
    
    \textbf{Answer:} $\forall x$ $Politician(x)\Rightarrow [\exists y$ $\forall t$ $Person(y)\Rightarrow Fools(x,y,t)] \land [\forall y$ $\exists t$ $Person(y)\Rightarrow Fools(x,y,t)] \land \neg[\forall y$ $\forall t$ $Person(y)\Rightarrow Fools(x,y,t)]$
    \item All Greeks speak the same language. (Use $Speaks(x,l)$ to mean that person $x$ speaks language $l$.)
    
    \textbf{Answer:} $\forall x,y,l$ $Greek(x)\land Greek(y) \land Speaks(x,l)\Rightarrow Speaks(y,l)$
\end{enumerate}



\end{homeworkProblem}

\begin{homeworkProblem}
\textit{(8 pts.)} For each pair of atomic sentences, give the most general unifier if it exists:

\begin{enumerate}[label=\Alph*.]

    \item $P(A,B,B), P(x,y,z)$.
    
    \textbf{Answer:} $\{x/A, y/B, z/B\}$
    
    Unification:
    
    $P(A,B,B), P(x,y,z): \{x/A\}$
    
    $P(A,B,B), P(A,y,z): \{x/A, y/B\} $
    
    $P(A,B,B), P(A,B,z): \{x/A, y/B, z/B\} $
    
    $P(A,B,B), P(A,B,B)$
    
    \item $Q(y,G(A,B)),Q(G(x,x),y)$.
    
    \textbf{Answer:} Do not exist
    
    Unification:
    
    $Q(y,G(A,B)),Q(G(x,x),y): \{y/G(x,x)\}$
    
    $Q(G(x,x),G(A,B)),Q(G(x,x),G(x,x)): \{y/G(x,x), x/A\}$
    
    $Q(G(A,A),G(A,B)),Q(G(A,A),G(A,A)): \{y/G(x,x), x/A\}$ Cannot unify $A/B$
    
    \item $Older(Father(y),y),Older(Father(x),John)$.
    
    \textbf{Answer:} $\{x/John, y/John\}$
    
    Unification:
    
    $Older(Father(y),y),Older(Father(x),John):$ need recursion
    
    \qquad $Older(Father(y), y), Older(Father(x), John) : \{x/y\}$
    
    $Older(Father(y), y), Older(Father(y), John) : \{x/y, y/John\} = \{x/John, y/John\}$
    
    $Older(Father(John), John), Older(Father(John), John)$
    
    \item $Knows(Father(y),y), Knows(x,x)$.

    \textbf{Answer:} Do not exist
    
    Unification:
    
    $Knows(Father(y),y), Knows(x,x): \{x/Father(y)\}$
    
    $Knows(Father(y),y), Knows(Father(y),Father(y)): \{x/Father(y)\}$ Cannot unify $y/Father(y)$
    
    
\end{enumerate}

\end{homeworkProblem}
